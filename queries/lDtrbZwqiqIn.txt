SET statement_timeout TO 1000;
SELECT M.content, (length(M.content) - length(replace(M.content, ' ', '')) + 1) wordage FROM "transcriptAnalyzer_message" M

INNER JOIN "transcriptAnalyzer_username" UN
  On
    M.user_id = UN.user_id

WHERE
    M.onebox = FALSE
  AND
    M.date = '2016-07-27'
  AND
    UN.name = 'El''endia Starman'

ORDER BY M.time;
