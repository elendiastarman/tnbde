SET statement_timeout TO 1000;
SELECT M.content, UN.name FROM "transcriptAnalyzer_message" M

INNER JOIN "transcriptAnalyzer_user" U

INNER JOIN "transcriptAnalyzer_username" UN

WHERE
    M.onebox = FALSE
  AND
    M.date = '2016-07-27'
  AND
    M.content LIKE '%ಠ_ಠ%'
  AND
    M.user_id = UN.user_id

ORDER BY M.time;
